Definition preimage {X Y} : (X ⇾ Y) ⇾ 𝒫 Y ⇾ 𝒫 X := set:(λ (f:X ⇾ Y) (V: 𝒫 Y), { x : X | f x ∊ V } ).

Record subset_el `(U:𝒫 X) :=
{ subset_pt :> set_T X
; subset_pt_is_el : subset_pt ∊ U

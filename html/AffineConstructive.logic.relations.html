Lemma Transitive_rel_proper_aimpl {A} R {H:@Transitive A R}
  {x₁ x₂} {y₁ y₂} : R x₂ x₁ → R y₁ y₂ → R x₁ y₁ ⊸ R x₂ y₂.
Proof. intros Ex Ey.
  rew <-(transitivity R x₂ x₁ y₂), (aprod_true_l Ex).
  rew <-(transitivity R x₁ y₁ y₂), (aprod_true_r Ey).
  refl.
Qed.

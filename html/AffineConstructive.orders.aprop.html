Lemma AProp_lattice_order : LatticeOrder Ω.
Proof. split.
+ split; [ exact _ |..].
  × exact aandl.
  × exact aandr.
  × intros P Q R. rew (aprod_aand _ _). exact (aand_lub _ _ _).
+ apply Build_JoinSemiLatticeOrder.
  × exact aorl.
  × exact aorr.
  × intros P Q R. rew (aprod_aand _ _). exact (aor_glb _ _ _).
Qed.

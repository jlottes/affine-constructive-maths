Lemma compose_pointed {X Y Z:set} {x:X} {y:Y} {z:Z} {g f} : Pointed_Morphism x y f → Pointed_Morphism y z g
  → Pointed_Morphism x z (g ∘ f).
Proof. intros. change (g (f x) = z). rew (preserves_point f). exact (preserves_point g). Qed.
